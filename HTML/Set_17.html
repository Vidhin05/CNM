A = 4
B = 0.5
C = 3
D = 1/3

def f1(x,y1,y2):
  return A*y1*(1-B*y2)

def f2(x,y1,y2):
  return C*y2*(D*y1-1)

hrange = [0.01,0.005]

y1_values = []
y2_values = []
x_values = []

y1_init = 3
y2_init = 3
x_init = 0

for h in hrange:
  ans_y1 = []
  ans_y2 = []
  y1 = y1_init
  y2 = y2_init
  ans_y1.append(y1)
  ans_y2.append(y2)
  xrange = np.arange(0,4,h)
  for x in xrange:
    v11 = f1(x,y1,y2)
    v12 = f2(x,y1,y2)
    
    v21 = f1(x + h/2 , y1 + h*v11/2 , y2 + h*v12/2)
    v22 = f2(x + h/2 , y1 + h*v11/2 , y2 + h*v12/2)
    
    v31 = f1(x + h/2 , y1 + h*v21/2 , y2 + h*v22/2)
    v32 = f2(x + h/2 , y1 + h*v21/2 , y2 + h*v22/2)
 
    v41 = f1(x + h , y1 + h*v31 ,  y2 + h*v32)
    v42 = f2(x + h , y1 + h*v31 ,  y2 + h*v32)
    
    y1 = y1 + h* (v11 + 2*v21 + 2*v31 + v41)/6 
    y2 = y2 + h* (v12 + 2*v22 + 2*v32 + v42)/6 
    
    ans_y1.append(y1)
    ans_y2.append(y2)
  y1_values.append(ans_y1[:-1])
  y2_values.append(ans_y2[:-1])
  x_values.append(xrange)
  
plt.plot(x_values[0],y1_values[0],label="h = 0.01", color='green')
plt.plot(x_values[1],y1_values[1],label="h = 0.005",color='#1f77b4')
plt.legend(loc="upper right")
plt.xlabel('$x$')
plt.ylabel('$y1$')
plt.title('Value of $y1$')
plt.grid(True)
plt.show()
plt.plot(x_values[0],y2_values[0],label="h = 0.01", color='green')
plt.plot(x_values[1],y2_values[1],label="h = 0.005",color='#1f77b4')
plt.legend(loc="upper right")
plt.xlabel('$x$')
plt.ylabel('$y2$')
plt.title('Value of $y2$')
plt.grid(True)
plt.show()
plt.plot(y1_values[0],y2_values[0], color='green')
plt.plot(y1_values[1],y2_values[1],color='#1f77b4')
plt.title('$Y2$ v/s $Y1$')
plt.xlabel('$Y1$')
plt.grid(True)
plt.ylabel('$Y2$')
plt.show()
